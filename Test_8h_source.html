362extern "C" void getBestThreshold(REAL *relThresh, REAL *score_pos,
363 REAL *score_neg) {
364 REAL interval = 0.01;
365 REAL min_score, max_score, bestThresh, tmpThresh, bestAcc, tmpAcc;
366 INT n_interval, correct, total;
367 for (INT r = 0; r < relationTotal; r++) {
368 if (validLef[r] == -1)
369 continue;
370 total = (validRig[r] - validLef[r] + 1) * 2;
371 min_score = score_pos[validLef[r]];
372 if (score_neg[validLef[r]] < min_score)
373 min_score = score_neg[validLef[r]];
374 max_score = score_pos[validLef[r]];
375 if (score_neg[validLef[r]] > max_score)
376 max_score = score_neg[validLef[r]];
377 for (INT i = validLef[r] + 1; i <= validRig[r]; i++) {
378 if (score_pos[i] < min_score)
379 min_score = score_pos[i];
380 if (score_pos[i] > max_score)
381 max_score = score_pos[i];
382 if (score_neg[i] < min_score)
383 min_score = score_neg[i];
384 if (score_neg[i] > max_score)
385 max_score = score_neg[i];
386 }
387 n_interval = INT((max_score - min_score) / interval);
388 for (INT i = 0; i <= n_interval; i++) {
389 tmpThresh = min_score + i * interval;
390 correct = 0;
391 for (INT j = validLef[r]; j <= validRig[r]; j++) {
392 if (score_pos[j] <= tmpThresh)
393 correct++;
394 if (score_neg[j] > tmpThresh)
395 correct++;
396 }
397 tmpAcc = 1.0 * correct / total;
398 if (i == 0) {
399 bestThresh = tmpThresh;
400 bestAcc = tmpAcc;
401 } else if (tmpAcc > bestAcc) {
402 bestAcc = tmpAcc;
403 bestThresh = tmpThresh;
404 }
405 }
406 relThresh[r] = bestThresh;
407 }
408}
